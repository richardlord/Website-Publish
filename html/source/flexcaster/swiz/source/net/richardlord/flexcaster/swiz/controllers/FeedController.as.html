package net.richardlord.flexcaster.swiz.controllers
{
    import flash.events.IEventDispatcher;
    
    import mx.rpc.AsyncToken;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    
    import net.richardlord.flexcaster.swiz.delegates.LoadFeedDelegate;
    import net.richardlord.flexcaster.swiz.events.AlertEvent;
    import net.richardlord.flexcaster.swiz.model.Feed;
    
    import org.swizframework.utils.services.ServiceHelper;
    
    public class FeedController
    {
        [Dispatcher]public var dispatcher : IEventDispatcher;
        
        [Inject("loadFeedDelegate" )]
        public var feedDelegate : LoadFeedDelegate;
        
        [Inject]public var feed : Feed;
        
        [Mediate("FeedEvent.SHOW_FEED", properties="feed" )]
        public function loadFeed( feedToLoad : Feed ) : void
        {
            if ( feedToLoad )
            {
                feed.assign( feedToLoad );
                if ( !feedToLoad.loaded )
                {
                    var token : AsyncToken = feedDelegate.send( feedToLoad.url );
                    var serviceHelper:ServiceHelper = new ServiceHelper();
                    serviceHelper.executeServiceCall( token, feedLoaded, feedFailed, [ feedToLoad ] );
                }
            }
        }
        
        private function feedLoaded( event : ResultEvent, listFeed : Feed ) : void
        {
            listFeed.assign( event.result as Feed );
            feed.assign( listFeed );
        }
        
        private function feedFailed( event : FaultEvent ) : void
        {
            dispatcher.dispatchEvent( new AlertEvent( AlertEvent.SHOW_ALERT, "Internet Error",
                "Unable to load the selected feed. Please check your internet connection then try again." ) );
        }
    }
}
