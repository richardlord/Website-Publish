<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    width="360" height="480">

    <mx:Script>
        <![CDATA[
            import net.richardlord.flexcaster.swiz.events.FeedEvent;
            import net.richardlord.flexcaster.swiz.model.Feed;
            import mx.collections.ArrayCollection;
            
            [Inject("feeds" )]
            [Bindable]public var feeds : ArrayCollection;
            
            private function showFeed() : void
            {
                dispatchEvent( new FeedEvent( FeedEvent.SHOW_FEED, feedsList.selectedItem as Feed ) );
                feedsList.selectedIndex = -1;
            }
        ]]>
    </mx:Script>

    <mx:Label width="360"
        text="Flexcaster Swiz"
        fontSize="14" fontWeight="bold"/>

    <mx:List id="feedsList"
        y="30" width="360" height="400"
        dataProvider="{feeds}"
        change="showFeed()"
        itemRenderer="net.richardlord.flexcaster.swiz.view.FeedRenderer"/>

</mx:Canvas>
