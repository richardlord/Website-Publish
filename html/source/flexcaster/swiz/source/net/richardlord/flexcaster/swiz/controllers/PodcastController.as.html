package net.richardlord.flexcaster.swiz.controllers
{
    import flash.events.IEventDispatcher;
    
    import net.richardlord.flexcaster.swiz.model.Podcast;
    
    public class PodcastController
    {
        [Dispatcher]public var dispatcher : IEventDispatcher;
        
        [Inject]public var podcast : Podcast;
        
        [Mediate("PodcastEvent.SHOW_PODCAST", properties="podcast" )]
        public function showPodcast( podcastToShow : Podcast ) : void
        {
            podcast.assign( podcastToShow );
        }
    }
}
