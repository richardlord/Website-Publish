package net.richardlord.flexcaster.smartysignals.controller.impl
{
    import mx.controls.Alert;
    
    import net.richardlord.flexcaster.smartysignals.controller.FeedController;
    import net.richardlord.flexcaster.smartysignals.model.FeedManager;
    import net.richardlord.flexcaster.smartysignals.model.StateManager;
    import net.richardlord.flexcaster.smartysignals.model.constant.FlexcasterStates;
    import net.richardlord.flexcaster.smartysignals.model.vo.Feed;
    import net.richardlord.flexcaster.smartysignals.service.LoadFeedService;
    
    public class SimpleFeedController implements FeedController
    {
        
        [Inject]public var feedLoadService : LoadFeedService;
        
        [Inject]public var feedModel : FeedManager;
        
        [Inject]public var stateManager : StateManager;
        
        public function showFeed( feed : Feed ) : void
        {
            feedModel.feed = feed;
            stateManager.state = FlexcasterStates.SHOW_FEED_STATE;
            if ( !feed.loaded )
            {
                feedLoadService.feedLoadFailedSignal.add( loadFailed );
                feedLoadService.load( feed );
            }
        }
        
        private function loadFailed():void
        {
            Alert.show( "Unable to load the selected feed. Please check your internet connection then try again.", "Internet Error" );
        }
    }
}
