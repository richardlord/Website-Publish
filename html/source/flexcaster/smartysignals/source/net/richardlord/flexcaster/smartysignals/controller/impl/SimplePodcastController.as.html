package net.richardlord.flexcaster.smartysignals.controller.impl
{
    import net.richardlord.flexcaster.smartysignals.controller.PodcastController;
    import net.richardlord.flexcaster.smartysignals.model.PodcastManager;
    import net.richardlord.flexcaster.smartysignals.model.StateManager;
    import net.richardlord.flexcaster.smartysignals.model.constant.FlexcasterStates;
    import net.richardlord.flexcaster.smartysignals.model.vo.Podcast;
    
    public class SimplePodcastController implements PodcastController
    {
        
        [Inject]public var podcastModel : PodcastManager;
        
        [Inject]public var stateManager : StateManager;
        
        public function showPodcast( podcast : Podcast ) : void
        {
            podcastModel.podcast = podcast;
            stateManager.state = FlexcasterStates.SHOW_PODCAST_STATE;
        }
    }
}
