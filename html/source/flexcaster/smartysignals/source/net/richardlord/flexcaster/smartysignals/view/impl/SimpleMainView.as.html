package net.richardlord.flexcaster.smartysignals.view.impl
{
    import net.expantra.smartypants.Injector;
    import net.richardlord.flexcaster.smartysignals.model.StateManager;
    import net.richardlord.flexcaster.smartysignals.model.constant.FlexcasterStates;
    import net.richardlord.flexcaster.smartysignals.view.MainView;
    
    public class SimpleMainView implements MainView
    {
        public static const FEEDS_VIEW : int = 0;
        
        public static const FEED_VIEW : int = 1;
        
        public static const PODCAST_VIEW : int = 2;
        
        [Inject]public var stateManager : StateManager;
        
        [Bindable]public var currentView : int;
        
        [PostConstruct]public function postInjection( injector : Injector ) : void
        {
            stateManager.stateChangeSignal.add( stateChanged );
            stateChanged( stateManager.state );
        }
        
        public function stateChanged( state : String ) : void
        {
            switch ( state )
            {
                case FlexcasterStates.SHOW_FEEDS_STATE:
                    currentView = FEEDS_VIEW;
                    break;
                case FlexcasterStates.SHOW_FEED_STATE:
                    currentView = FEED_VIEW;
                    break;
                case FlexcasterStates.SHOW_PODCAST_STATE:
                    currentView = PODCAST_VIEW;
                    break;
            }
        }
    }
}
