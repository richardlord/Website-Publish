package net.richardlord.flexcaster.smartysignals.model.impl
{
    import net.richardlord.flexcaster.smartysignals.model.FeedManager;
    import net.richardlord.flexcaster.smartysignals.model.vo.Feed;
    
    import org.osflash.signals.ISignal;
    import org.osflash.signals.Signal;
    
    public class SimpleFeedManager implements FeedManager
    {
        private var _feed : Feed;
        
        private var _feedSignal : Signal;
        
        public function SimpleFeedManager()
        {
            _feedSignal = new Signal( Feed );
        }
        
        public function get feed() : Feed
        {
            return _feed;
        }
        
        public function set feed( value : Feed ) : void
        {
            _feed = value;
            _feedSignal.dispatch( _feed );
        }
        
        public function get feedChangeSignal() : ISignal
        {
            return _feedSignal;
        }
    }
}
