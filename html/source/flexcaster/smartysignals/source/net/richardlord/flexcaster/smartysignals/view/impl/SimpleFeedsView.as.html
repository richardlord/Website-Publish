package net.richardlord.flexcaster.smartysignals.view.impl
{
    import flash.events.Event;
    import flash.events.EventDispatcher;
    
    import net.expantra.smartypants.Injector;
    import net.richardlord.flexcaster.smartysignals.controller.FeedController;
    import net.richardlord.flexcaster.smartysignals.model.FeedsManager;
    import net.richardlord.flexcaster.smartysignals.model.vo.Feed;
    import net.richardlord.flexcaster.smartysignals.view.FeedsView;
    
    public class SimpleFeedsView extends EventDispatcher implements FeedsView
    {
        
        [Inject]public var feedsModel : FeedsManager
        
        [Inject]public var feedController : FeedController;
        
        [PostConstruct]public function setUpModel( injector : Injector ) : void
        {
            feedsModel.feedsChangeSignal.add( feedsChanged );
            dispatchEvent( new Event( "feedsChanged" ) );
        }
        
        [Bindable( event="feedsChanged" )]
        public function get feeds() : Array
        {
            return feedsModel.feeds;
        }
        
        private function feedsChanged( feeds : Array ) : void
        {
            dispatchEvent( new Event( "feedsChanged" ) );
        }
        
        public function showFeed( index : int ) : void
        {
            feedController.showFeed( feedsModel.feeds[ index ] as Feed );
        }
    }
}
