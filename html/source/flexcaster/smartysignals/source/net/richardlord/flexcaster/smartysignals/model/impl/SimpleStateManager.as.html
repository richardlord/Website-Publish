package net.richardlord.flexcaster.smartysignals.model.impl
{
    import net.richardlord.flexcaster.smartysignals.model.StateManager;
    
    import org.osflash.signals.ISignal;
    import org.osflash.signals.Signal;
    
    public class SimpleStateManager implements StateManager
    {
        private var _state : String;
        
        private var _stateSignal : Signal;
        
        public function SimpleStateManager()
        {
            _stateSignal = new Signal( String );
        }
        
        public function get state() : String
        {
            return _state;
        }
        
        public function set state( value : String ) : void
        {
            _state = value;
            _stateSignal.dispatch( _state );
        }
        
        public function get stateChangeSignal() : ISignal
        {
            return _stateSignal;
        }
    }
}
