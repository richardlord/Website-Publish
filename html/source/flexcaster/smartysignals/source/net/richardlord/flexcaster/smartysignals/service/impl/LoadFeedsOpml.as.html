package net.richardlord.flexcaster.smartysignals.service.impl
{
    import flash.events.Event;
    import flash.events.IOErrorEvent;
    import flash.net.URLLoader;
    import flash.net.URLRequest;
    
    import net.richardlord.flexcaster.smartysignals.model.vo.Feed;
    import net.richardlord.flexcaster.smartysignals.service.LoadFeedsService;
    
    import org.osflash.signals.ISignal;
    import org.osflash.signals.Signal;
    
    public class LoadFeedsOpml implements LoadFeedsService
    {
        private static const URL : String = "feeds.opml";
        
        private var _feedsLoadCompleteSignal : Signal;
        
        private var _feedsLoadFailedSignal : Signal;
        
        private var _loader : URLLoader;
        
        public function LoadFeedsOpml()
        {
            _feedsLoadCompleteSignal = new Signal( Array );
            _feedsLoadFailedSignal = new Signal();
        }
        
        public function get feedsLoadCompleteSignal() : ISignal
        {
            return _feedsLoadCompleteSignal;
        }
        
        public function get feedsLoadFailedSignal() : ISignal
        {
            return _feedsLoadFailedSignal;
        }
        
        public function load() : void
        {
            _loader = new URLLoader();
            _loader.addEventListener( Event.COMPLETE, parseLoadedData );
            _loader.addEventListener( IOErrorEvent.IO_ERROR, dispatchLoadError );
            _loader.addEventListener( Event.COMPLETE, cleanUpAfterLoad );
            _loader.addEventListener( IOErrorEvent.IO_ERROR, cleanUpAfterLoad );
            var request : URLRequest = new URLRequest( URL );
            _loader.load( request );
        }
        
        private function parseLoadedData( event : Event ) : void
        {
            var opml : XML = new XML( _loader.data );
            var feeds : Array = new Array();
            for each ( var outline : XML in opml.body.outline )
            {
                feeds.push( new Feed( outline.@text, outline.@xmlUrl ) );
            }
            _feedsLoadCompleteSignal.dispatch( feeds );
        }
        
        private function dispatchLoadError( event : Event ) : void
        {
            _feedsLoadFailedSignal.dispatch();
        }
        
        private function cleanUpAfterLoad( event : Event ) : void
        {
            _loader.removeEventListener( Event.COMPLETE, parseLoadedData );
            _loader.removeEventListener( IOErrorEvent.IO_ERROR, dispatchLoadError );
            _loader.removeEventListener( Event.COMPLETE, cleanUpAfterLoad );
            _loader.removeEventListener( IOErrorEvent.IO_ERROR, cleanUpAfterLoad );
            _loader = null;
        }
    }
}
