package net.richardlord.flexcaster.smartysignals.config
{
    import net.expantra.smartypants.Injector;
    import net.expantra.smartypants.SmartyPants;
    import net.richardlord.flexcaster.smartysignals.controller.*;
    import net.richardlord.flexcaster.smartysignals.controller.impl.*;
    import net.richardlord.flexcaster.smartysignals.model.*;
    import net.richardlord.flexcaster.smartysignals.model.impl.*;
    import net.richardlord.flexcaster.smartysignals.service.*;
    import net.richardlord.flexcaster.smartysignals.service.impl.*;
    import net.richardlord.flexcaster.smartysignals.view.*;
    import net.richardlord.flexcaster.smartysignals.view.impl.*;
    
    public class FlexcasterConfig
    {
        public function configure( document : Object ) : void
        {
            var injector : Injector = SmartyPants.getOrCreateInjectorFor( document );
            
            injector.newRule().whenAskedFor( FeedsManager ).useSingletonOf( SimpleFeedsManager );
            injector.newRule().whenAskedFor( FeedManager ).useSingletonOf( SimpleFeedManager );
            injector.newRule().whenAskedFor( PodcastManager ).useSingletonOf( SimplePodcastManager );
            injector.newRule().whenAskedFor( StateManager ).useSingletonOf( SimpleStateManager );
            
            injector.newRule().whenAskedFor( FeedsController ).useSingletonOf( SimpleFeedsController );
            injector.newRule().whenAskedFor( FeedController ).useSingletonOf( SimpleFeedController );
            injector.newRule().whenAskedFor( PodcastController ).useSingletonOf( SimplePodcastController );
            
            injector.newRule().whenAskedFor( LoadFeedsService ).createInstanceOf( LoadFeedsOpml );
            injector.newRule().whenAskedFor( LoadFeedService ).createInstanceOf( LoadFeedRss );
            
            injector.newRule().whenAskedFor( MainView ).createInstanceOf( SimpleMainView );
            injector.newRule().whenAskedFor( FeedsView ).createInstanceOf( SimpleFeedsView );
            injector.newRule().whenAskedFor( FeedView ).createInstanceOf( SimpleFeedView );
            injector.newRule().whenAskedFor( PodcastView ).createInstanceOf( SimplePodcastView );
        }
    }
}
