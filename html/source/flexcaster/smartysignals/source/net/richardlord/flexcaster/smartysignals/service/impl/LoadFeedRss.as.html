package net.richardlord.flexcaster.smartysignals.service.impl
{
    import flash.events.Event;
    import flash.events.IOErrorEvent;
    import flash.net.URLLoader;
    import flash.net.URLRequest;
    
    import net.richardlord.flexcaster.smartysignals.model.vo.Feed;
    import net.richardlord.flexcaster.smartysignals.model.vo.Podcast;
    import net.richardlord.flexcaster.smartysignals.service.LoadFeedService;
    
    import org.osflash.signals.ISignal;
    import org.osflash.signals.Signal;
    
    public class LoadFeedRss implements LoadFeedService
    {
        private var _feedLoadCompleteSignal : Signal;
        
        private var _feedLoadFailedSignal : Signal;
        
        private var _loader : URLLoader;
        
        private var feed : Feed;
        
        public function LoadFeedRss()
        {
            _feedLoadCompleteSignal = new Signal( Feed );
            _feedLoadFailedSignal = new Signal();
        }
        
        public function get feedLoadCompleteSignal() : ISignal
        {
            return _feedLoadCompleteSignal;
        }
        
        public function get feedLoadFailedSignal() : ISignal
        {
            return _feedLoadFailedSignal;
        }
        
        public function load( feed : Feed ) : void
        {
            this.feed = feed;
            _loader = new URLLoader();
            _loader.addEventListener( Event.COMPLETE, parseLoadedData );
            _loader.addEventListener( IOErrorEvent.IO_ERROR, dispatchLoadError );
            _loader.addEventListener( Event.COMPLETE, cleanUpAfterLoad );
            _loader.addEventListener( IOErrorEvent.IO_ERROR, cleanUpAfterLoad );
            var request : URLRequest = new URLRequest( feed.url );
            _loader.load( request );
        }
        
        private function parseLoadedData( event : Event ) : void
        {
            var rss : XML = new XML( _loader.data );
            var itunes : Namespace = new Namespace( "http://www.itunes.com/dtds/podcast-1.0.dtd" );
            var channel : XML = rss.channel[ 0 ];
            feed.title = channel.title;
            feed.url = channel.link;
            feed.description = channel.itunes::summary;
            if ( !feed.description )
            {
                feed.description = channel.description;
            }
            feed.image = channel.itunes::image.@href;
            
            feed.podcasts = new Array();
            for each ( var item : XML in channel.item )
            {
                var podcast : Podcast = new Podcast();
                podcast.title = item.title;
                podcast.description = item.itunes::summary;
                if ( !podcast.description )
                {
                    podcast.description = item.description;
                }
                podcast.url = item.enclosure.@url;
                podcast.length = item.enclosure.@length;
                podcast.pubDate = new Date( Date.parse( item.pubDate.toString() ) );
                feed.podcasts.push( podcast );
            }
            
            feed.loaded = true;
            _feedLoadCompleteSignal.dispatch( feed );
        }
        
        private function dispatchLoadError( event : Event ) : void
        {
            _feedLoadFailedSignal.dispatch();
        }
        
        private function cleanUpAfterLoad( event : Event ) : void
        {
            _loader.removeEventListener( Event.COMPLETE, parseLoadedData );
            _loader.removeEventListener( IOErrorEvent.IO_ERROR, dispatchLoadError );
            _loader.removeEventListener( Event.COMPLETE, cleanUpAfterLoad );
            _loader.removeEventListener( IOErrorEvent.IO_ERROR, cleanUpAfterLoad );
            _loader = null;
        }
    }
}
