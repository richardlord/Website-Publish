package net.richardlord.flexcaster.smartysignals.view.impl
{
    import flash.events.Event;
    import flash.events.EventDispatcher;
    
    import net.expantra.smartypants.Injector;
    import net.richardlord.flexcaster.smartysignals.controller.FeedController;
    import net.richardlord.flexcaster.smartysignals.model.FeedManager;
    import net.richardlord.flexcaster.smartysignals.model.PodcastManager;
    import net.richardlord.flexcaster.smartysignals.model.vo.Feed;
    import net.richardlord.flexcaster.smartysignals.model.vo.Podcast;
    import net.richardlord.flexcaster.smartysignals.view.PodcastView;
    
    public class SimplePodcastView extends EventDispatcher implements PodcastView
    {
        
        [Inject]public var feedModel : FeedManager;
        
        [Inject]public var podcastModel : PodcastManager;
        
        [Inject]public var feedController : FeedController;
        
        [PostConstruct]public function setUpModel( injector : Injector ) : void
        {
            feedModel.feedChangeSignal.add( feedChanged );
            dispatchEvent( new Event( "feedChanged" ) );
            podcastModel.podcastChangeSignal.add( podcastChanged );
            dispatchEvent( new Event( "podcastChanged" ) );
        }
        
        private function feedChanged( feed : Feed ) : void
        {
            dispatchEvent( new Event( "feedChanged" ) );
        }
        
        private function podcastChanged( podcast : Podcast ) : void
        {
            dispatchEvent( new Event( "podcastChanged" ) );
        }
        
        [Bindable( event="feedChanged" )]
        public function get feed() : Feed
        {
            return feedModel.feed;
        }
        
        [Bindable("podcastChanged" )]
        public function get podcast() : Podcast
        {
            return podcastModel.podcast;
        }
        
        public function showFeed() : void
        {
            feedController.showFeed( feedModel.feed );
        }
    }
}
