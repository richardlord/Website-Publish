package net.richardlord.flexcaster.smartysignals.controller.impl
{
    import mx.core.IMXMLObject;
    
    import net.expantra.smartypants.Injector;
    import net.expantra.smartypants.SmartyPants;
    import net.richardlord.flexcaster.smartysignals.config.FlexcasterConfig;
    import net.richardlord.flexcaster.smartysignals.controller.FeedsController;
    
    public class StartupController implements IMXMLObject
    {
        
        [Inject]public var controller : FeedsController;
        
        public function initialized( document : Object, id : String ) : void
        {
            var config : FlexcasterConfig = new FlexcasterConfig();
            config.configure( document );
            var injector : Injector = SmartyPants.getOrCreateInjectorFor( document );
            injector.injectInto( this );
            controller.loadFeeds();
        }
    }
}
