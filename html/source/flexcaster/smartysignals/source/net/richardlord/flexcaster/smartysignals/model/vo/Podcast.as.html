package net.richardlord.flexcaster.smartysignals.model.vo
{
    
    [Bindable]public class Podcast
    {
        public var title : String;
        
        public var url : String;
        
        public var pubDate : Date;
        
        public var description : String;
        
        public var length : int;
        
        public function assign( other : Podcast ) : void
        {
            title = other.title;
            url = other.url;
            pubDate = other.pubDate;
            description = other.description;
            length = other.length;
        }
    }
}
