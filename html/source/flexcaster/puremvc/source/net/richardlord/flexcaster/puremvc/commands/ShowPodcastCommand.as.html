package net.richardlord.flexcaster.puremvc.commands
{
    import net.richardlord.flexcaster.puremvc.model.PodcastProxy;
    import net.richardlord.flexcaster.puremvc.model.vo.Podcast;
    import net.richardlord.flexcaster.puremvc.notifications.NotificationNames;
    
    import org.puremvc.as3.interfaces.INotification;
    import org.puremvc.as3.patterns.command.SimpleCommand;
    
    public class ShowPodcastCommand extends SimpleCommand
    {
        override public function execute( notification : INotification ) : void
        {
            var podcast : Podcast = notification.getBody() as Podcast;
            var podcastProxy : PodcastProxy = facade.retrieveProxy( PodcastProxy.NAME ) as PodcastProxy;
            podcastProxy.podcast = podcast;
            sendNotification( NotificationNames.DISPLAY_PODCAST );
        }
    }
}
