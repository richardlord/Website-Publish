package net.richardlord.flexcaster.puremvc.model.business
{
    import mx.rpc.AsyncToken;
    import mx.rpc.IResponder;
    import mx.rpc.Responder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    import mx.rpc.http.HTTPService;
    
    import net.richardlord.flexcaster.puremvc.model.vo.Feed;
    
    public class LoadFeedsDelegate
    {
        private var _service : HTTPService;
        
        private var _token : AsyncToken;
        
        public function LoadFeedsDelegate()
        {
            _service = new HTTPService();
            _service.url = "feeds.opml";
            _service.resultFormat = "e4x";
        }
        
        public function send() : AsyncToken
        {
            var localResponder : Responder = new Responder( feedsLoaded, loadFailed );
            var token : AsyncToken = _service.send();
            token.addResponder( localResponder );
            _token = new AsyncToken( token.message );
            return _token;
        }
        
        private function feedsLoaded( event : ResultEvent ) : void
        {
            var opml : XML = event.result as XML;
            var feeds : Array = new Array();
            for each ( var outline : XML in opml.body.outline )
            {
                feeds.push( new Feed( outline.@text, outline.@xmlUrl ) );
            }
            var resultEvent : ResultEvent = new ResultEvent( ResultEvent.RESULT, false, true, feeds, _token );
            for each ( var responder : IResponder in _token.responders )
            {
                responder.result( resultEvent );
            }
        }
        
        private function loadFailed( event : FaultEvent ) : void
        {
            for each ( var responder : IResponder in _token.responders )
            {
                responder.fault( event );
            }
        }
    }
}
