package net.richardlord.flexcaster.puremvc.model
{
    import mx.rpc.IResponder;
    
    import net.richardlord.flexcaster.puremvc.model.vo.Podcast;
    
    import org.puremvc.as3.interfaces.IProxy;
    import org.puremvc.as3.patterns.proxy.Proxy;
    
    public class PodcastProxy extends Proxy implements IProxy
    {
        public static const NAME : String = "PodcastProxy";
        
        public function PodcastProxy( data : Object = null )
        {
            super( NAME, data );
        }
        
        public function get podcast() : Podcast
        {
            return data as Podcast;
        }
        
        public function set podcast( value : Podcast ) : void
        {
            data = value;
        }
    }
}
