package net.richardlord.flexcaster.puremvc.view
{
    import flash.events.Event;
    import flash.events.MouseEvent;
    
    import mx.events.FlexEvent;
    import mx.events.ListEvent;
    
    import net.richardlord.flexcaster.puremvc.model.FeedProxy;
    import net.richardlord.flexcaster.puremvc.model.vo.AlertData;
    import net.richardlord.flexcaster.puremvc.notifications.NotificationNames;
    import net.richardlord.flexcaster.puremvc.view.components.FeedView;
    
    import org.puremvc.as3.interfaces.IMediator;
    import org.puremvc.as3.interfaces.INotification;
    import org.puremvc.as3.patterns.mediator.Mediator;
    
    public class FeedMediator extends Mediator implements IMediator
    {
        public static const NAME : String = "FeedMediator";
        
        private var feedProxy : FeedProxy;
        
        public function get feedView() : FeedView
        {
            return viewComponent as FeedView;
        }
        
        public function FeedMediator( viewComponent : FeedView )
        {
            super( NAME, viewComponent );
            feedProxy = facade.retrieveProxy( FeedProxy.NAME ) as FeedProxy;
            feedView.addEventListener( FlexEvent.CREATION_COMPLETE, completeHandler );
        }
        
        private function completeHandler( event : FlexEvent ) : void
        {
            feedView.addEventListener( FlexEvent.SHOW, showHandler );
            feedView.podcastsList.addEventListener( ListEvent.CHANGE, podcastSelected );
            feedView.back.addEventListener( MouseEvent.CLICK, backToFeeds );
            showHandler( event );
        }
        
        private function showHandler( event : FlexEvent = null ) : void
        {
            if ( feedProxy.feed )
            {
                feedView.title.text = feedProxy.feed.title;
                feedView.podcastsList.dataProvider = feedProxy.feed.podcasts;
            }
        }
        
        override public function listNotificationInterests() : Array
        {
            return [
                NotificationNames.LOAD_FEED_SUCCESS,
                NotificationNames.LOAD_FEED_FAILURE
                ];
        }
        
        override public function handleNotification( note : INotification ) : void
        {
            switch ( note.getName() )
            {
                case NotificationNames.LOAD_FEED_SUCCESS:
                    feedView.title.text = feedProxy.feed.title;
                    feedView.podcastsList.dataProvider = feedProxy.feed.podcasts;
                    break;
                case NotificationNames.LOAD_FEED_FAILURE:
                    sendNotification( NotificationNames.SHOW_ALERT, new AlertData( "Network Error", "Unable to load the selected feed. Please check your internet connection then try again." ) );
                    break;
            }
        }
        
        private function podcastSelected( event : Event ) : void
        {
            sendNotification( NotificationNames.SHOW_PODCAST, feedView.podcastsList.selectedItem );
            feedView.podcastsList.selectedIndex = -1;
        }
        
        private function backToFeeds( event : Event ) : void
        {
            sendNotification( NotificationNames.SHOW_FEEDS );
        }
    }
}
