package net.richardlord.flexcaster.puremvc.model.vo
{
    
    [Bindable]public class Feed
    {
        public var title : String;
        
        public var url : String;
        
        public var description : String;
        
        public var image : String;
        
        public var podcasts : Array;
        
        public var loaded : Boolean = false;
        
        public function Feed( title : String = "", url : String = "" )
        {
            this.title = title;
            this.url = url;
            loaded = false;
            podcasts = new Array();
        }
        
        public function assign( other : Feed ) : void
        {
            title = other.title;
            url = other.url;
            description = other.description;
            image = other.image;
            podcasts = other.podcasts;
            loaded = other.loaded;
        }
    }
}
