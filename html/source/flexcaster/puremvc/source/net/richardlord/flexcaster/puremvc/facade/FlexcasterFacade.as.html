package net.richardlord.flexcaster.puremvc.facade
{
    import mx.core.Application;
    
    import net.richardlord.flexcaster.puremvc.commands.ShowFeedCommand;
    import net.richardlord.flexcaster.puremvc.commands.ShowFeedsCommand;
    import net.richardlord.flexcaster.puremvc.commands.ShowAlertCommand;
    import net.richardlord.flexcaster.puremvc.commands.ShowPodcastCommand;
    import net.richardlord.flexcaster.puremvc.commands.StartupCommand;
    import net.richardlord.flexcaster.puremvc.notifications.NotificationNames;
    
    import org.puremvc.as3.patterns.facade.Facade;
    
    public class FlexcasterFacade extends Facade
    {
        public static function getInstance() : FlexcasterFacade
        {
            if ( instance == null )
            {
                instance = new FlexcasterFacade();
            }
            return instance as FlexcasterFacade;
        }
        
        override protected function initializeController() : void
        {
            super.initializeController();
            registerCommand( NotificationNames.STARTUP, StartupCommand );
            registerCommand( NotificationNames.SHOW_FEEDS, ShowFeedsCommand );
            registerCommand( NotificationNames.SHOW_FEED, ShowFeedCommand );
            registerCommand( NotificationNames.SHOW_PODCAST, ShowPodcastCommand );
            registerCommand( NotificationNames.SHOW_ALERT, ShowAlertCommand );
        }
        
        public function startup( app : Application ) : void
        {
            sendNotification( NotificationNames.STARTUP, app );
            sendNotification( NotificationNames.SHOW_FEEDS );
        }
    }
}
