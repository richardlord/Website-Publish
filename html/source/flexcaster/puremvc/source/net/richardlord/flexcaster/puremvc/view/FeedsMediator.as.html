package net.richardlord.flexcaster.puremvc.view
{
    import flash.events.Event;
    
    import mx.collections.ArrayCollection;
    import mx.events.ListEvent;
    
    import net.richardlord.flexcaster.puremvc.model.FeedsProxy;
    import net.richardlord.flexcaster.puremvc.model.vo.AlertData;
    import net.richardlord.flexcaster.puremvc.notifications.NotificationNames;
    import net.richardlord.flexcaster.puremvc.view.components.FeedsView;
    
    import org.puremvc.as3.interfaces.IMediator;
    import org.puremvc.as3.interfaces.INotification;
    import org.puremvc.as3.patterns.mediator.Mediator;
    
    public class FeedsMediator extends Mediator implements IMediator
    {
        public static const NAME : String = "FeedsMediator";
        
        private var feedsProxy : FeedsProxy;
        
        public function get feedsView() : FeedsView
        {
            return viewComponent as FeedsView;
        }
        
        public function FeedsMediator( viewComponent : FeedsView )
        {
            super( NAME, viewComponent );
            feedsProxy = facade.retrieveProxy( FeedsProxy.NAME ) as FeedsProxy;
            feedsView.feedsList.addEventListener( ListEvent.CHANGE, feedSelected );
        }
        
        override public function listNotificationInterests() : Array
        {
            return [
                NotificationNames.LOAD_FEEDS_SUCCESS,
                NotificationNames.LOAD_FEEDS_FAILURE,
                ];
        }
        
        override public function handleNotification( note : INotification ) : void
        {
            switch ( note.getName() )
            {
                case NotificationNames.LOAD_FEEDS_SUCCESS:
                    feedsView.feedsList.dataProvider = new ArrayCollection( feedsProxy.feeds );
                    break;
                case NotificationNames.LOAD_FEEDS_FAILURE:
                    sendNotification( NotificationNames.SHOW_ALERT, new AlertData( "Network Error", "Unable to load the list of feeds. Please check your internet connection then reload the page." ) );
                    break;
            }
        }
        
        private function feedSelected( event : Event ) : void
        {
            sendNotification( NotificationNames.SHOW_FEED, feedsView.feedsList.selectedItem );
            feedsView.feedsList.selectedIndex = -1;
        }
    }
}
