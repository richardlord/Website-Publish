package net.richardlord.flexcaster.puremvc.view
{
    import flash.events.Event;
    import flash.events.MouseEvent;
    
    import mx.events.FlexEvent;
    
    import net.richardlord.flexcaster.puremvc.model.FeedProxy;
    import net.richardlord.flexcaster.puremvc.model.PodcastProxy;
    import net.richardlord.flexcaster.puremvc.notifications.NotificationNames;
    import net.richardlord.flexcaster.puremvc.view.components.PodcastView;
    
    import org.puremvc.as3.interfaces.IMediator;
    import org.puremvc.as3.patterns.mediator.Mediator;
    
    public class PodcastMediator extends Mediator implements IMediator
    {
        public static const NAME : String = "PodcastMediator";
        
        private var podcastProxy : PodcastProxy;
        
        private var feedProxy : FeedProxy;
        
        public function get podcastView() : PodcastView
        {
            return viewComponent as PodcastView;
        }
        
        public function PodcastMediator( viewComponent : PodcastView )
        {
            super( NAME, viewComponent );
            
            podcastProxy = facade.retrieveProxy( PodcastProxy.NAME ) as PodcastProxy;
            feedProxy = facade.retrieveProxy( FeedProxy.NAME ) as FeedProxy;
            podcastView.addEventListener( FlexEvent.CREATION_COMPLETE, completeHandler );
        }
        
        private function completeHandler( event : FlexEvent ) : void
        {
            podcastView.addEventListener( FlexEvent.SHOW, showHandler );
            podcastView.addEventListener( FlexEvent.HIDE, hideHandler );
            podcastView.back.addEventListener( MouseEvent.CLICK, backToFeed );
            showHandler( event );
        }
        
        private function showHandler( event : FlexEvent ) : void
        {
            if ( podcastProxy.podcast && feedProxy.feed )
            {
                podcastView.player.url = podcastProxy.podcast.url;
                podcastView.title.text = podcastProxy.podcast.title;
                podcastView.description.text = podcastProxy.podcast.description;
                podcastView.image.source = feedProxy.feed.image;
                podcastView.player.play();
            }
        }
        
        private function hideHandler( event : FlexEvent ) : void
        {
            podcastView.player.pause();
        }
        
        private function backToFeed( event : Event ) : void
        {
            sendNotification( NotificationNames.SHOW_FEED );
        }
    }
}
