package net.richardlord.flexcaster.puremvc.commands
{
    import net.richardlord.flexcaster.puremvc.model.FeedProxy;
    import net.richardlord.flexcaster.puremvc.model.vo.Feed;
    import net.richardlord.flexcaster.puremvc.notifications.NotificationNames;
    
    import org.puremvc.as3.interfaces.INotification;
    import org.puremvc.as3.patterns.command.SimpleCommand;
    
    public class ShowFeedCommand extends SimpleCommand
    {
        override public function execute( notification : INotification ) : void
        {
            var feed : Feed = notification.getBody() as Feed;
            if ( feed )
            {
                var feedProxy : FeedProxy = facade.retrieveProxy( FeedProxy.NAME ) as FeedProxy;
                feedProxy.feed = feed;
            }
            sendNotification( NotificationNames.DISPLAY_FEED );
        }
    }
}
