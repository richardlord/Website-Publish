<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:media="net.richardlord.flexcaster.flex3.media.*"
    width="360" height="480"
    hide="hideHandler()" show="showHandler()">

    <mx:Script>
        <![CDATA[
            import net.richardlord.flexcaster.flex3.events.FeedEvent;
            import net.richardlord.flexcaster.flex3.data.Feed;
            import net.richardlord.flexcaster.flex3.data.Podcast;
            
            [Bindable]
            public var podcast : Podcast;
            
            [Bindable]
            public var feed : Feed;
            
            private function showHandler() : void
            {
                player.play();
            }
            
            private function hideHandler() : void
            {
                player.pause();
            }
            
            private function goBack() : void
            {
                dispatchEvent( new FeedEvent( FeedEvent.SHOW_FEED ) );
            }
        ]]>
    </mx:Script>

    <media:Mp3Player id="player"
        url="{podcast.url}"/>

    <mx:Label id="title"
        width="360"
        text="{podcast.title}"
        fontSize="14" fontWeight="bold"/>

    <mx:Image y="30" width="320" height="240"
        source="{feed.image}"/>

    <mx:HSlider id="slider"
        y="280" width="360"
        liveDragging="true" maximum="{player.duration}" minimum="0" value="{player.position}"
        change="player.position=slider.value"/>

    <mx:Button y="300"
        label="Play"
        right="0"
        click="player.play()" visible="{!player.playing}"/>
    <mx:Button y="300"
        label="Pause"
        right="0"
        click="player.pause()" visible="{player.playing}"/>

    <mx:TextArea id="description"
        y="330" width="360" height="100"
        htmlText="{podcast.description}"
        editable="false"/>

    <mx:Button id="back"
        y="440"
        label="Back"
        click="goBack()"/>
</mx:Canvas>
