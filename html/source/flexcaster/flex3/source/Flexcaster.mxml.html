<?xml version="1.0" encoding="utf-8"?>
<mx:Application 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:views="net.richardlord.flexcaster.flex3.views.*"
    width="400" height="500"
    layout="absolute"
    creationComplete="creationCompleteHandler()"
    viewSourceURL="/source/flexcaster/flex3">

    <mx:Script>
        <![CDATA[
            import net.richardlord.flexcaster.flex3.events.PodcastEvent;
            import net.richardlord.flexcaster.flex3.events.FeedEvent;
            import net.richardlord.flexcaster.flex3.events.SimpleEvent;
            
            private function creationCompleteHandler() : void
            {
                addEventListener( SimpleEvent.SHOW_FEEDS, showFeedsHandler );
                addEventListener( FeedEvent.SHOW_FEED, showFeedHandler );
                addEventListener( PodcastEvent.SHOW_PODCAST, showPodcastHandler );
            }
            
            private function showFeedsHandler( event : SimpleEvent ) : void
            {
                views.selectedIndex = 0;
            }
            
            private function showFeedHandler( event : FeedEvent ) : void
            {
                views.selectedIndex = 1;
                if ( event.feed )
                {
                    feedView.feed = event.feed;
                }
            }
            
            private function showPodcastHandler( event : PodcastEvent ) : void
            {
                views.selectedIndex = 2;
                if ( event.podcast )
                {
                    podcastView.feed = feedView.feed;
                    podcastView.podcast = event.podcast;
                }
            }
        ]]>
    </mx:Script>

    <mx:ViewStack id="views"
        x="20" y="10">
        <views:FeedsView id="feedsView"/>
        <views:FeedView id="feedView"/>
        <views:PodcastView id="podcastView"/>
    </mx:ViewStack>

</mx:Application>
