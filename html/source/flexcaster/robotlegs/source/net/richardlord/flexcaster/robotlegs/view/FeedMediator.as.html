package net.richardlord.flexcaster.robotlegs.view
{
    import flash.events.MouseEvent;
    
    import mx.events.ListEvent;
    
    import net.richardlord.flexcaster.robotlegs.events.FeedEvent;
    import net.richardlord.flexcaster.robotlegs.events.FeedsEvent;
    import net.richardlord.flexcaster.robotlegs.events.PodcastEvent;
    import net.richardlord.flexcaster.robotlegs.models.FlexcasterModel;
    import net.richardlord.flexcaster.robotlegs.models.vo.Feed;
    import net.richardlord.flexcaster.robotlegs.models.vo.Podcast;
    import net.richardlord.flexcaster.robotlegs.view.components.FeedView;
    
    import org.robotlegs.mvcs.Mediator;
    
    public class FeedMediator extends Mediator
    {
        [Inject]public var feedView:FeedView;
        [Inject]public var model:FlexcasterModel;
        
        private var feed:Feed;
        
        override public function onRegister():void
        {
            eventMap.mapListener( eventDispatcher, FeedEvent.FEED_LOADED, onFeedLoaded );
            eventMap.mapListener( feedView.podcastsList, ListEvent.CHANGE, onPodcastSelected );
            eventMap.mapListener( feedView.back, MouseEvent.CLICK, onBackSelected );
            
            setFeed( model.feed );
        }
        
        private function onFeedLoaded( event:FeedEvent ):void
        {
            if( event.feed )
            {
                setFeed( event.feed );
            }
        }
        
        private function setFeed( feed:Feed ):void
        {
            this.feed = feed;
            feedView.title.text = feed.title;
            feedView.podcastsList.dataProvider = feed.podcasts;
        }
        
        private function onPodcastSelected( event:ListEvent ):void
        {
            dispatch( new PodcastEvent( PodcastEvent.DISPLAY_PODCAST, feedView.podcastsList.selectedItem as Podcast, feed ) );
            feedView.podcastsList.selectedIndex = -1;
        }
        
        private function onBackSelected( event:MouseEvent ):void
        {
            feedView.podcastsList.dataProvider = null;
            dispatch( new FeedsEvent( FeedsEvent.DISPLAY_FEEDS ) );
        }
    }
}
