package net.richardlord.flexcaster.robotlegs.view
{
    import net.richardlord.flexcaster.robotlegs.events.FeedEvent;
    import net.richardlord.flexcaster.robotlegs.events.FeedsEvent;
    import net.richardlord.flexcaster.robotlegs.events.PodcastEvent;
    import net.richardlord.flexcaster.robotlegs.view.components.MainView;
    
    import org.robotlegs.mvcs.Mediator;
    
    public class MainViewMediator extends Mediator
    {
        [Inject]public var mainView:MainView;
        
        override public function onRegister():void
        {
            eventMap.mapListener( eventDispatcher, FeedsEvent.DISPLAY_FEEDS, onDisplayFeeds );
            eventMap.mapListener( eventDispatcher, FeedEvent.DISPLAY_FEED, onDisplayFeed );
            eventMap.mapListener( eventDispatcher, PodcastEvent.DISPLAY_PODCAST, onDisplayPodcast );
        }
        
        private function onDisplayFeeds( event:FeedsEvent ):void
        {
            mainView.views.selectedIndex = 0;
        }
        
        private function onDisplayFeed( event:FeedEvent ):void
        {
            mainView.views.selectedIndex = 1;
        }
        
        private function onDisplayPodcast( event:PodcastEvent ):void
        {
            mainView.views.selectedIndex = 2;
        }
    }
}
