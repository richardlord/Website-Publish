package net.richardlord.flexcaster.robotlegs.services
{
    import mx.rpc.AsyncToken;
    import mx.rpc.Responder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    import mx.rpc.http.HTTPService;
    
    import net.richardlord.flexcaster.robotlegs.events.AlertEvent;
    import net.richardlord.flexcaster.robotlegs.events.FeedEvent;
    import net.richardlord.flexcaster.robotlegs.models.vo.Feed;
    import net.richardlord.flexcaster.robotlegs.models.vo.Podcast;
    
    import org.robotlegs.mvcs.Actor;

    public class RssFeedService extends Actor implements IFeedService
    {
        private var feed:Feed;
        
        public function loadFeed( feed:Feed ):void
        {
            this.feed = feed;
            var service:HTTPService = new HTTPService();
            service.url = feed.url;
            service.resultFormat = "e4x";
            var token : AsyncToken = service.send();
            token.addResponder( new Responder( feedLoaded, loadFailed ) );
        }
        
        private function feedLoaded( event : ResultEvent ) : void
        {
            var itunes : Namespace = new Namespace( "http://www.itunes.com/dtds/podcast-1.0.dtd" );
            var channel : XML = ( event.result as XML ).channel[ 0 ];
            feed.title = channel.title;
            feed.url = channel.link;
            feed.description = channel.itunes::summary;
            if ( !feed.description )
            {
                feed.description = channel.description;
            }
            feed.image = channel.itunes::image.@href;
            
            feed.podcasts = new Array();
            for each ( var item : XML in channel.item )
            {
                var podcast : Podcast = new Podcast();
                podcast.title = item.title;
                podcast.description = item.itunes::summary;
                if ( !podcast.description )
                {
                    podcast.description = item.description;
                }
                podcast.url = item.enclosure.@url;
                podcast.format = item.enclosure.@type;
                podcast.length = item.enclosure.@length;
                podcast.pubDate = new Date( Date.parse( item.pubDate.toString() ) );
                feed.podcasts.push( podcast );
            }
            
            feed.loaded = true;
            dispatch( new FeedEvent( FeedEvent.FEED_LOADED, feed ) );
        }
        
        private function loadFailed( event : FaultEvent ) : void
        {
            dispatch( new AlertEvent( AlertEvent.SHOW_ALERT, "Internet Error",
                "Unable to load the selected feed. Please check your internet connection then try again." ) );
        }
    }
}
