package net.richardlord.flexcaster.robotlegs.contexts
{
    import flash.display.DisplayObjectContainer;
    
    import net.richardlord.flexcaster.robotlegs.commands.*;
    import net.richardlord.flexcaster.robotlegs.events.*;
    import net.richardlord.flexcaster.robotlegs.models.FlexcasterModel;
    import net.richardlord.flexcaster.robotlegs.services.*;
    import net.richardlord.flexcaster.robotlegs.view.*;
    import net.richardlord.flexcaster.robotlegs.view.components.*;
    
    import org.robotlegs.mvcs.Context;
    
    public class FlexcasterContext extends Context
    {
        public function FlexcasterContext()
        {
            super();
        }
        
        override public function startup():void
        {
            commandMap.mapEvent( FeedsEvent.LOAD_FEEDS, LoadFeedsCommand, FeedsEvent );
            commandMap.mapEvent( FeedsEvent.FEEDS_LOADED, SetFeedsCommand, FeedsEvent );
            commandMap.mapEvent( FeedEvent.DISPLAY_FEED, LoadFeedCommand, FeedEvent );
            commandMap.mapEvent( FeedEvent.DISPLAY_FEED, SetFeedCommand, FeedEvent );
            commandMap.mapEvent( PodcastEvent.DISPLAY_PODCAST, SetPodcastCommand, PodcastEvent );
            commandMap.mapEvent( AlertEvent.SHOW_ALERT, ShowAlertCommand, AlertEvent );

            injector.mapSingleton( FlexcasterModel );
            injector.mapClass( IFeedsService, OpmlFeedsService );
            injector.mapClass( IFeedService, RssFeedService );
            
            mediatorMap.mapView( MainView, MainViewMediator );
            mediatorMap.mapView( FeedsView, FeedsMediator );
            mediatorMap.mapView( FeedView, FeedMediator );
            mediatorMap.mapView( PodcastView, PodcastMediator );
            
            dispatchEvent( new FeedsEvent( FeedsEvent.LOAD_FEEDS ) );
            dispatchEvent( new FeedsEvent( FeedsEvent.DISPLAY_FEEDS ) );
        }
    }
}
