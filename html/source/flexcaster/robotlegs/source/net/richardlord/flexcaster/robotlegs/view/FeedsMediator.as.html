package net.richardlord.flexcaster.robotlegs.view
{
    import mx.events.ListEvent;
    
    import net.richardlord.flexcaster.robotlegs.events.FeedEvent;
    import net.richardlord.flexcaster.robotlegs.events.FeedsEvent;
    import net.richardlord.flexcaster.robotlegs.models.FlexcasterModel;
    import net.richardlord.flexcaster.robotlegs.models.vo.Feed;
    import net.richardlord.flexcaster.robotlegs.view.components.FeedsView;
    
    import org.robotlegs.mvcs.Mediator;
    
    public class FeedsMediator extends Mediator
    {
        [Inject]public var feedsView:FeedsView;
        [Inject]public var model:FlexcasterModel;
        
        override public function onRegister():void
        {
            eventMap.mapListener( eventDispatcher, FeedsEvent.FEEDS_CHANGED, onFeedsChanged );
            eventMap.mapListener( feedsView.feedsList, ListEvent.CHANGE, onFeedSelected );
            
            feedsView.feedsList.dataProvider = model.feeds;
        }
        
        private function onFeedsChanged( event:FeedsEvent ):void
        {
            feedsView.feedsList.dataProvider = event.feeds;
        }
        
        private function onFeedSelected( event:ListEvent ):void
        {
            dispatch( new FeedEvent( FeedEvent.DISPLAY_FEED, feedsView.feedsList.selectedItem as Feed ) );
            feedsView.feedsList.selectedIndex = -1;
        }
    }
}
