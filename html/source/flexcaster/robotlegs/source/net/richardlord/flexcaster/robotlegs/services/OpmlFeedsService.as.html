package net.richardlord.flexcaster.robotlegs.services
{
    import mx.rpc.AsyncToken;
    import mx.rpc.Responder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    import mx.rpc.http.HTTPService;
    
    import net.richardlord.flexcaster.robotlegs.events.AlertEvent;
    import net.richardlord.flexcaster.robotlegs.events.FeedsEvent;
    import net.richardlord.flexcaster.robotlegs.models.vo.Feed;
    
    import org.robotlegs.mvcs.Actor;

    public class OpmlFeedsService extends Actor implements IFeedsService
    {
        private static const URL:String = "feeds.opml";
        
        public function loadFeeds():void
        {
            var service:HTTPService = new HTTPService();
            service.url = URL;
            service.resultFormat = "e4x";
            var token : AsyncToken = service.send();
            token.addResponder( new Responder( feedsLoaded, loadFailed ) );
        }
        
        private function feedsLoaded( event : ResultEvent ) : void
        {
            var opml : XML = event.result as XML;
            var feeds : Array = new Array();
            for each ( var outline : XML in opml.body.outline )
            {
                feeds.push( new Feed( outline.@text, outline.@xmlUrl ) );
            }
            dispatch( new FeedsEvent( FeedsEvent.FEEDS_LOADED, feeds ) );
        }
        
        private function loadFailed( event : FaultEvent ) : void
        {
            dispatch( new AlertEvent( AlertEvent.SHOW_ALERT, "Internet Error",
                "Unable to load the list of feeds. Please check your internet connection then reload the page." ) );
        }
    }
}
