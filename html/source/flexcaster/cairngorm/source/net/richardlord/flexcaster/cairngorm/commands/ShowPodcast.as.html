package net.richardlord.flexcaster.cairngorm.commands
{
    import com.adobe.cairngorm.commands.ICommand;
    import com.adobe.cairngorm.control.CairngormEvent;
    
    import net.richardlord.flexcaster.cairngorm.model.FlexcasterModelLocator;
    import net.richardlord.flexcaster.cairngorm.model.Podcast;
    
    public class ShowPodcast implements ICommand
    {
        private var _model : FlexcasterModelLocator;
        
        public function ShowPodcast()
        {
            _model = FlexcasterModelLocator.getInstance()
        }
        
        public function execute( event : CairngormEvent ) : void
        {
            _model.selectedView = Flexcaster.PODCAST_VIEW;
            var podcast : Podcast = event.data as Podcast;
            if ( podcast )
            {
                _model.podcast = podcast;
            }
        }
    }
}
