package net.richardlord.flexcaster.cairngorm.commands
{
    import com.adobe.cairngorm.commands.ICommand;
    import com.adobe.cairngorm.control.CairngormEvent;
    
    import mx.collections.ArrayCollection;
    import mx.rpc.AsyncToken;
    import mx.rpc.IResponder;
    import mx.rpc.events.ResultEvent;
    
    import net.richardlord.flexcaster.cairngorm.delegates.LoadFeedsDelegate;
    import net.richardlord.flexcaster.cairngorm.events.AlertEvent;
    import net.richardlord.flexcaster.cairngorm.model.FlexcasterModelLocator;
    
    public class ShowFeeds implements ICommand, IResponder
    {
        private var _model : FlexcasterModelLocator = FlexcasterModelLocator.getInstance();
        
        public function execute( event : CairngormEvent ) : void
        {
            _model.selectedView = Flexcaster.FEEDS_VIEW;
            if ( !_model.feeds )
            {
                var delegate : LoadFeedsDelegate = new LoadFeedsDelegate();
                var token : AsyncToken = delegate.send();
                token.addResponder( this );
            }
        }
        
        public function result( event : Object ) : void
        {
            var feeds : Array = ResultEvent( event ).result as Array;
            if ( feeds )
            {
                _model.feeds = new ArrayCollection( feeds );
            }
        }
        
        public function fault( event : Object ) : void
        {
            var alertEvent : AlertEvent = new AlertEvent( AlertEvent.ALERT,
                "Internet Error",
                "Unable to load the list of feeds. Please check your internet connection then reload the page." );
            alertEvent.dispatch();
        }
    }
}
