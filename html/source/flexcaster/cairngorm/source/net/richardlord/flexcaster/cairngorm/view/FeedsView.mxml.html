<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    width="360" height="480">

    <mx:Script>
        <![CDATA[
            import net.richardlord.flexcaster.cairngorm.events.SimpleEvent;
            import net.richardlord.flexcaster.cairngorm.model.Feed;
            import net.richardlord.flexcaster.cairngorm.model.FlexcasterModelLocator;
            
            [Bindable]
            public var _model : FlexcasterModelLocator = FlexcasterModelLocator.getInstance();
            
            private function showFeed() : void
            {
                var event : SimpleEvent = new SimpleEvent( SimpleEvent.SHOW_FEED, feedsList.selectedItem );
                event.dispatch();
                feedsList.selectedIndex = -1;
            }
        ]]>
    </mx:Script>

    <mx:Label width="360"
        text="Flexcaster Cairngorm"
        fontSize="14" fontWeight="bold"/>

    <mx:List id="feedsList"
        y="30" width="360" height="400"
        dataProvider="{_model.feeds}"
        change="showFeed()"
        itemRenderer="net.richardlord.flexcaster.cairngorm.view.FeedRenderer"/>

</mx:Canvas>
