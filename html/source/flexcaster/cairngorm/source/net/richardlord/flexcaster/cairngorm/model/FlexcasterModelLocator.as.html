package net.richardlord.flexcaster.cairngorm.model
{
    import com.adobe.cairngorm.CairngormError;
    import com.adobe.cairngorm.CairngormMessageCodes;
    import com.adobe.cairngorm.model.IModelLocator;
    
    import mx.collections.ArrayCollection;
    
    public class FlexcasterModelLocator implements IModelLocator
    {
        private static var _instance : FlexcasterModelLocator;
        
        public static function getInstance() : FlexcasterModelLocator
        {
            if ( _instance == null )
            {
                _instance = new FlexcasterModelLocator();
            }
            return _instance;
        }
        
        [Bindable]public var feeds : ArrayCollection;
        
        [Bindable]public var feed : Feed;
        
        [Bindable]public var podcast : Podcast;
        
        [Bindable]public var selectedView : int;
        
        public function FlexcasterModelLocator()
        {
            if ( _instance != null )
            {
                throw new CairngormError(
                    CairngormMessageCodes.SINGLETON_EXCEPTION, "PodcastModelLocator" );
            }
            _instance = this;
        }
    }
}
