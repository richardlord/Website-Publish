package net.richardlord.flexcaster.parsley.actions
{
    import flash.events.EventDispatcher;
    
    import mx.rpc.AsyncToken;
    import mx.rpc.Responder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    
    import net.richardlord.flexcaster.parsley.delegates.LoadFeedDelegate;
    import net.richardlord.flexcaster.parsley.events.AlertEvent;
    import net.richardlord.flexcaster.parsley.events.FeedEvent;
    import net.richardlord.flexcaster.parsley.model.Feed;
    
    [Event("showAlert",type="net.richardlord.flexcaster.parsley.events.AlertEvent")]
    
    [ManagedEvents("showAlert")]

    public class ShowFeedAction extends EventDispatcher
    {
        [Inject]public var feedDelegate : LoadFeedDelegate;
        
        [Inject]public var feed : Feed;
        
        private var selectedFeed : Feed;
        
        [MessageHandler("showFeed")]
        public function loadFeed( event : FeedEvent ) : void
        {
            if ( event.feed )
            {
                selectedFeed = event.feed;
                feed.assign( event.feed );
                if ( !event.feed.loaded )
                {
                    var token : AsyncToken = feedDelegate.send( feed.url );
                    token.addResponder( new Responder( feedLoaded, feedFailed ) );
                }
            }
        }
        
        private function feedLoaded( event : ResultEvent ) : void
        {
            selectedFeed.assign( event.result as Feed );
            feed.assign( selectedFeed );
        }
        
        private function feedFailed( event : FaultEvent ) : void
        {
            dispatchEvent( new AlertEvent( AlertEvent.SHOW_ALERT, "Internet Error",
                "Unable to load the selected feed. Please check your internet connection then try again." ) );
        }
    }
}
