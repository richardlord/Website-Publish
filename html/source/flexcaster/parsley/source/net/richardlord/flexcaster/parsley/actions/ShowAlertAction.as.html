package net.richardlord.flexcaster.parsley.actions
{
    import mx.controls.Alert;
    
    import net.richardlord.flexcaster.parsley.events.AlertEvent;

    public class ShowAlertAction
    {
        [MessageHandler("showAlert" )]
        public function showAlert( event : AlertEvent ) : void
        {
            Alert.show( event.message, event.title );
        }
    }
}
