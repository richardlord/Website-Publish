package net.richardlord.flexcaster.parsley.actions
{
    import flash.events.EventDispatcher;
    
    import mx.collections.ArrayCollection;
    import mx.rpc.AsyncToken;
    import mx.rpc.Responder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    
    import net.richardlord.flexcaster.parsley.delegates.LoadFeedsDelegate;
    import net.richardlord.flexcaster.parsley.events.AlertEvent;
    import net.richardlord.flexcaster.parsley.events.SimpleEvent;
    
    [ManagedEvents("showAlert,showFeeds")]
    [Event("showFeeds",type="net.richardlord.flexcaster.parsley.events.SimpleEvent")]
    [Event("showAlert",type="net.richardlord.flexcaster.parsley.events.AlertEvent")]

    public class LoadFeedsAction extends EventDispatcher
    {
        [Inject]public var feedsDelegate : LoadFeedsDelegate;
        
        [Inject("feeds" )]
        public var feeds : ArrayCollection;
        
        [Init]public function loadFeeds() : void
        {
            dispatchEvent( new SimpleEvent( SimpleEvent.SHOW_FEEDS ) );
            if ( feeds.length == 0 )
            {
                var token : AsyncToken = feedsDelegate.send();
                token.addResponder( new Responder( feedsLoaded, feedsFailed ) );
            }
        }
        
        private function feedsLoaded( ev : ResultEvent ) : void
        {
            feeds.source = ev.result as Array;
        }
        
        private function feedsFailed( ev : FaultEvent ) : void
        {
            dispatchEvent( new AlertEvent( AlertEvent.SHOW_ALERT, "Internet Error",
                "Unable to load the list of feeds. Please check your internet connection then reload the page." ) );
        }
    }
}
