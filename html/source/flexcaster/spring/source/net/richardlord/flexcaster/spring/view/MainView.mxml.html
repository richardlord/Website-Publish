<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    xmlns:view="net.richardlord.flexcaster.spring.view.*"
    width="400" height="500">

    <mx:ViewStack id="views"
        x="20" y="10">
        <view:FeedsView id="feedsView"/>
        <view:FeedView id="feedView"/>
        <view:PodcastView id="podcastView"/>
    </mx:ViewStack>

    <mx:Script>
        <![CDATA[
            import mx.controls.Alert;
            
            import net.richardlord.flexcaster.spring.events.AlertEvent;
            import net.richardlord.flexcaster.spring.events.FeedEvent;
            import net.richardlord.flexcaster.spring.events.PodcastEvent;
            import net.richardlord.flexcaster.spring.events.SimpleEvent;
            
            import org.springextensions.actionscript.core.event.EventBus;
            
            [EventHandler("showAlert" )]
            public function alertEventHandler( event : AlertEvent ) : void
            {
                Alert.show( event.message, event.title );
            }
            
            [EventHandler("showFeeds" )]
            public function showFeedsEventHandler( event:Event = null ) : void
            {
                views.selectedIndex = 0;
            }
            
            [EventHandler("showFeed" )]
            public function showFeedEventHandler( event:Event = null ) : void
            {
                views.selectedIndex = 1;
            }
            
            [EventHandler("showPodcast" )]
            public function showPodcastEventHandler( event:Event = null ) : void
            {
                views.selectedIndex = 2;
            }
        ]]>
    </mx:Script>
