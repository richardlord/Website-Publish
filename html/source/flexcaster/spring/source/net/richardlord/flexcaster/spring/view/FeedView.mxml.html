<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml"
    width="360" height="480">

    <mx:Metadata>
        [RouteEvents("showFeeds,showPodcast")]
        [Event("showFeeds",type="net.richardlord.flexcaster.spring.events.SimpleEvent")]
        [Event("showPodcast",type="net.richardlord.flexcaster.spring.events.PodcastEvent")]
    </mx:Metadata>

    <mx:Script>
        <![CDATA[
            import net.richardlord.flexcaster.spring.events.PodcastEvent;
            import net.richardlord.flexcaster.spring.events.SimpleEvent;
            import net.richardlord.flexcaster.spring.model.Feed;
            import net.richardlord.flexcaster.spring.model.Podcast;
            
            [Autowired]
Bindable]public var feed : Feed;
            
            private function showPodcast() : void
            {
                dispatchEvent( new PodcastEvent( PodcastEvent.SHOW_PODCAST, podcastsList.selectedItem as Podcast ) );
                podcastsList.selectedIndex = -1;
            }
            
            private function goBack() : void
            {
                dispatchEvent( new SimpleEvent( SimpleEvent.SHOW_FEEDS ) );
            }
        ]]>
    </mx:Script>

    <mx:Label id="title"
        width="360"
        text="{feed.title}"
        fontSize="14" fontWeight="bold"/>

    <mx:List id="podcastsList"
        y="30" width="360" height="400"
        dataProvider="{feed.podcasts}"
        change="showPodcast()"
        itemRenderer="net.richardlord.flexcaster.spring.view.PodcastRenderer"/>

    <mx:Button id="back"
        y="440"
        label="Back"
        click="goBack()"/>
</mx:Canvas>
