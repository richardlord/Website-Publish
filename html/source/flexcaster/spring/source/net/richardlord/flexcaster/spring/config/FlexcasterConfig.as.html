package net.richardlord.flexcaster.spring.config
{
    import flash.events.EventDispatcher;
    
    import mx.core.Application;
    import mx.core.IMXMLObject;
    import mx.events.FlexEvent;
    
    import net.richardlord.flexcaster.spring.controllers.FeedController;
    import net.richardlord.flexcaster.spring.controllers.FeedsController;
    import net.richardlord.flexcaster.spring.controllers.PodcastController;
    import net.richardlord.flexcaster.spring.events.SimpleEvent;
    
    import org.springextensions.actionscript.context.support.FlexXMLApplicationContext;
    import org.springextensions.actionscript.core.event.EventBus;
    import org.springextensions.actionscript.ioc.factory.config.EventHandlerMetaDataPostProcessor;
    import org.springextensions.actionscript.ioc.factory.config.RouteEventsMetaDataPostProcessor;
    import org.springextensions.actionscript.stage.DefaultAutowiringStageProcessor;

    public class FlexcasterConfig extends EventDispatcher implements IMXMLObject
    {
        // Force inclusion of classes not referenced elsewhere in the code
        {
            FeedsController,FeedController,PodcastController,EventHandlerMetaDataPostProcessor,DefaultAutowiringStageProcessor,RouteEventsMetaDataPostProcessor;
        }
        
        [Bindable]
Embed("config.xml", mimeType ="application/octet-stream")]
        public var contextConfig:Class;
        
        public var applicationContext:FlexXMLApplicationContext;
        
        public function initialized( document:Object, id:String ):void
        {
            var component:Application = document as Application;
            if( component )
            {
                component.addEventListener( FlexEvent.APPLICATION_COMPLETE, createContext );
            }
            else
            {
                throw new Error( "The config should be included in the Application class." );
            }
        }
            
        private function createContext( event:FlexEvent ) : void
        {
            applicationContext = new FlexXMLApplicationContext();
            applicationContext.addEmbeddedConfig( contextConfig );
            applicationContext.load();
            startApp();
        }
        
        public function startApp() : void
        {
            EventBus.dispatchEvent( new SimpleEvent( SimpleEvent.SHOW_FEEDS ) );
        }
    }
}
